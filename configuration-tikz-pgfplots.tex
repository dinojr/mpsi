\RequirePackage{pgfplots}
\pgfkeys{/pgf/number format/.cd, set decimal separator={,}, 1000 sep={ }}

\pgfplotsset{compat=newest}

\pgfplotsset{%
  colormap = {tango}{color=(DarkSkyBlue) color=(LightPlum)}}

\colorlet{ExtraTickColor}{DarkPlum}

\pgfplotsset{extra tick style={color=ExtraTickColor}
}

\pgfplotscreateplotcyclelist{monochrome}{%
        {black},%
        {black,dashed},%
        {black,densely dotted},%
        {DarkGray},%
        {DarkGray,dashed},%
        {DarkGray,densely dotted},%             
        {Gray,dashed},%
        }

\pgfplotscreateplotcyclelist{polychrome}{%
        {SkyBlue},%
        {Plum},%
        {Chameleon},%
        {SkyBlue,dashed},%
        {Plum,dashed},%
        {Chameleon,dashed},%
        {SkyBlue,dotted},%
        {Plum,dotted},%             
        {Chameleon,dotted},%
        }       

\pgfplotsset{%
  petit graphe/.style = {%
    width = 7cm},
  tres petit graphe/.style = {%
    width = 6cm},
}    

\pgfplotsset{%
  courbe coin/.style = {%
    every axis y label/.append style=      
      {at={(axis description cs:-0.1,.5)}},
    every axis x label/.append style=
       {at={(axis description cs:1,.1)}},
    axis x line = bottom,
    axis y line = left,
    axis line style={-Latex[round]}
       }
}

\pgfplotsset{%
  courbe centre/.style = {%
    every axis y label/.append style=      
      {at={(axis description cs:0.1,1)}},
    every axis x label/.append style=
       {at={(axis description cs:1,.1)}},
    axis lines=center,
    axis line style={-Latex[round]}
       }
}

\pgfplotsset{%
  courbe boite/.style = {%
    every axis y label/.append style=      
      {at={(axis description cs:-0.1,.5)}},
    every axis x label/.append style=
       {at={(axis description cs:.5,-.1)}},
    axis lines = box,
       }
}

\pgfplotsset{every axis plot/.append style = {thick},
             every axis plot post/.append style = {mark = none},}

\pgfplotsset{points de mesure/.style= {%
                    every axis plot post/.append style = {%
                    mark = +,     
                    only marks,
    }}}
