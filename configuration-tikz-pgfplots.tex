\RequirePackage{pgfplots}
\pgfkeys{/pgf/number format/.cd, set decimal separator={,}, 1000 sep={ }}

\pgfplotsset{compat=newest}

\pgfplotsset{%
  colormap = {tango}{color=(DarkSkyBlue) color=(LightPlum)}}

\colorlet{ExtraTickColor}{DarkPlum}

\pgfplotsset{extra tick style={color=ExtraTickColor}
}

\pgfplotscreateplotcyclelist{monochrome}{%
        {black},%
        {black,dashed},%
        {black,densely dotted},%
        {DarkGray},%
        {DarkGray,dashed},%
        {DarkGray,densely dotted},%             
        {Gray,dashed},%
        }

\pgfplotscreateplotcyclelist{polychrome}{%
        {SkyBlue},%
        {Plum},%
        {Chameleon},%
        {SkyBlue,dashed},%
        {Plum,dashed},%
        {Chameleon,dashed},%
        {SkyBlue,dotted},%
        {Plum,dotted},%             
        {Chameleon,dotted},%
        }       

\pgfplotsset{%
  petit graphe/.style = {%
    width = 7cm},
  tres petit graphe/.style = {%
    width = 6cm},
}    

\pgfplotsset{%
  courbe coin/.style = {%
    axis x line = bottom,
    axis y line = left,
    near ticklabel align=inside,
    every axis y label/.append style=      
      {at={(ticklabel cs:0.5)},anchor=near yticklabel},
    every axis x label/.append style=
       {at={(ticklabel cs:1)},anchor = near xticklabel, },
    axis line style={-Latex[round]}
       }
}

\pgfplotsset{%
  courbe centre/.style = {%
    axis lines=center,
    near ticklabel align=center,
    every axis y label/.append style=      
      {% at={(axis description cs:0.5,1)}, 
      anchor=south},
    every axis x label/.append style=
       {at={(ticklabel cs:1)}, anchor=near xticklabel},
    axis line style={-Latex[round]}
       }
}

\pgfplotsset{%
  axis lines = box,       
  courbe boite/.style = {%
    every axis y label/.append style=      
      {at={(ticklabel cs:.5)},anchor = near yticklabel},
    every axis x label/.append style=
       {at={(ticklabel cs:.5)}, anchor = near xticklabel},
       }
}
