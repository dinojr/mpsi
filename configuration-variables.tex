\def\maketitle{%
  \par
  \begingroup
  \if@twocolumn
  \twocolumn[\@maketitle]
  \else \newpage
  \global\@topnum\z@ 
  \@maketitle
  \fi
  \endgroup
  \setcounter{footnote}{0}
  \let\maketitle\relax%
  \let\@maketitle\relax%
}

\def\@maketitle{%

  \end{center}
  \par
  \vskip 1.5em
}
