\RequirePackage{mathtools}
% pour DeclarePairedDelimiter et co.

\def\cste{\mathrm{cste}}

\protected\def\ProdVec{%
  \@ifnextchar*{\ProdVec@i}{\ProdVec@ii}}
\def\ProdVec@i*#1#2{%
  #1 \wedge #2}
\def\ProdVec@ii#1#2{%
  \vv{#1} \wedge \vv{#2}}
\protected\def\ProdScal{%
  \@ifnextchar*{\ProdScal@i}{\ProdScal@ii}}
\def\ProdScal@i*#1#2{%
  #1 \cdot #2}
\def\ProdScal@ii#1#2{%
  \vv{#1} \cdot \vv{#2}}
